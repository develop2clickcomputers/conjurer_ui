export interface PhoneNumbers {
    country_code: string;
    area_code: string;
    number: number,
    editable_mode: boolean;
    show_buttons: boolean;
}

/**
 * Address
 */
export interface Address {
    unit_no: any;
    address1: any;
    address2: any;
    city: string;
    state: string;
    country: string;
    pin_code: number;
    editable_mode: boolean;
    show_buttons: boolean;
}

/** Contact */
export interface Contact {
    id: number;
    carrier_name: string;
    carrier_abbr: string;
    email: any;
    addresses: Address[];
    phone_numbers: PhoneNumbers[];
}
